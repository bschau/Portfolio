NPM: Make packages local to user

mkdir ~/.npm-global
npm config set prefix '~/.npm-global'

... add to profile:

NPM_PACKAGES="$HOME/.npm-global"
PATH="$NPM_PACKAGES/bin:$PATH"
export NPM_PACKAGES PATH
