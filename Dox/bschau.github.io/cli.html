OpenSSL

  • PASSWORD is your current password
  • TEMPPASS is a temporary password only used in the conversion
  • PFXIn is the file you want to convert
  • PFXOut is the target file for the PKCS12 without passphrase

Remove password from .PFX file

openssl pkcs12 -clcerts -nokeys -in $PFXIn -out certificate.crt -password pass:$PASSWORD -passin pass:$PASSWORD               # extract certificate
openssl pkcs12 -cacerts -nokeys -in $PFXIn -out ca-cert.ca -password pass:$PASSWORD -passin pass:$PASSWORD                    # extract CA certificate
openssl pkcs12 -nocerts -in $PFXIn -out private.key -password pass:$PASSWORD -passin pass:$PASSWORD -passout pass:$TEMPPASS   # extract private key
openssl rsa -in private.key -out new.key -passin pass:$TEMPPASS                                                               # remove pass-phrase
cat new.key certificate.crt ca-cert.ca > pem.pem                                                                              # collect everything ...
openssl pkcs12 -export -nodes -CAfile ca-cert.ca -in pem.pem -out $PFXOut -passout pass:                                      # ... and create PKCS12 file
rm -f ca-cert.ca certificate.crt $PFXIn private.key new.key pem.pem                                                           # cleanup

Add extra intermediate

openssl pkcs12 -clcerts -nokeys -in $PFXIn -out certificate.crt -password pass:$PASSWORD -passin pass:$PASSWORD               # extract certificate
openssl pkcs12 -cacerts -nokeys -in $PFXIn -out ca-cert.ca -password pass:$PASSWORD -passin pass:$PASSWORD                    # extract CA certificate
openssl pkcs12 -nocerts -in $PFXIn -out private.key -password pass:$PASSWORD -passin pass:$PASSWORD -passout pass:$PASSWORD   # extract private key
cat extended.cer ca-cert.ca > chain.crt                                                                                       # create chain
openssl pkcs12 -export -out $PFXOut -inkey private.key -in certificate.crt -certfile chain.crt                                # create PKCS12 file
rm -f extended.cer ca-cert.ca certificate.crt $PFXIn private.key                                                              # cleanup
