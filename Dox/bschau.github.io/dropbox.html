Dropbox

Install on Ubuntu 20.04, server

sudo apt update
sudo apt -y install wget libglapi-mesa libxdamage1 libxfixes3 libxcb-glx0 libxcb-dri2-0
sudo apt -y install libxcb-dri3-0 libxcb-present0 libxcb-sync1 libxshmfence1 libxxf86vm1
cd ~ && wget -O - "https://www.dropbox.com/download?plat=lnx.x86_64" | tar xzf - ~/.dropbox-dist/dropboxd
sudo wget -O /usr/local/bin/dropbox "https://www.dropbox.com/download?dl=packages/dropbox.py"
sudo chmod 755 /usr/local/bin/dropbox
dropbox lansync n

To autostart Dropbox, create the file /etc/systemd/system/dropbox.service with following content:

[Unit]
Description=Dropbox Service
After=network.target

[Service]
ExecStart=/bin/sh -c '/usr/local/bin/dropbox start'
ExecStop=/bin/sh -c '/usr/local/bin/dropbox stop'
PIDFile=${HOME}/.dropbox/dropbox.pid
User=bs
Group=bs
Type=forking
Restart=on-failure
RestartSec=5
StartLimitInterval=60s
StartLimitBurst=3

[Install]
WantedBy=multi-user.target

... then execute:

sudo systemctl daemon-reload
sudo systemctl enable dropbox
sudo systemctl start dropbox
